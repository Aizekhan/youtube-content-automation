## Quick Fixes:

### 1. Update content.html

```javascript
function populateCTA(item) {
    const ctaData = item.cta_data || {};
    const ctaSegments = ctaData.cta_segments || [];

    if (!ctaSegments || ctaSegments.length === 0) {
        document.getElementById('ctaContent').innerHTML = '<div class="alert alert-info">...</div>';
        return;
    }

    let html = `<h6>CTA Segments (${ctaSegments.length})</h6>`;

    ctaSegments.forEach((segment, idx) => {
        const ctaAudio = segment.cta_audio_segment || {};

        html += `
            <div class="scene-card mb-3">
                <div class="d-flex justify-content-between align-items-start mb-2">
                    <div>
                        <span class="badge bg-primary">${segment.type || 'CTA'}</span>
                        <span class="badge bg-secondary ms-1">${segment.position || 'N/A'}</span>
                    </div>
                </div>

                <div style="background: rgba(102, 126, 234, 0.1); padding: 1rem; border-radius: 8px; mb-2">
                    <p class="mb-0" style="font-size: 1.05rem; color: #ffffff;">${segment.cta_text || 'No text'}</p>
                </div>

                ${segment.style_note ? `
                    <p class="text-muted mb-2"><small><i class="bi bi-info-circle"></i> ${segment.style_note}</small></p>
                ` : ''}

                ${ctaAudio.s3_url ? `
                    <div class="mt-2" style="background: rgba(40, 167, 69, 0.1); padding: 0.75rem; border-radius: 6px;">
                        <h6 style="font-size: 0.9rem; margin-bottom: 0.5rem;">🎤 Audio Track</h6>
                        <p class="mb-1" style="font-size: 0.85rem;">
                            <strong>Voice:</strong> ${ctaAudio.voice_id || 'N/A'}
                            <span class="text-muted">(${ctaAudio.voice_profile || 'N/A'})</span>
                        </p>
                        <p class="mb-1" style="font-size: 0.85rem;">
                            <strong>Duration:</strong> ${ctaAudio.duration_ms ? (ctaAudio.duration_ms / 1000).toFixed(1) + 's' : 'N/A'}
                        </p>
                        <p class="mb-0" style="font-size: 0.85rem;">
                            <strong>Cost:</strong> $${ctaAudio.generation_cost_usd ? ctaAudio.generation_cost_usd.toFixed(4) : 'N/A'}
                        </p>
                    </div>
                ` : `
                    <div class="alert alert-warning mt-2">
                        ⚠️ Audio not generated
                    </div>
                `}
            </div>
        `;
    });

    document.getElementById('ctaContent').innerHTML = html;
}
```
